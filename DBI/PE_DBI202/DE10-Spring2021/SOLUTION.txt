USE DE10
CREATE DATABASE DE10
CREATE TABLE CUSTOMERS(
SSN VARCHAR(20) PRIMARY KEY,
NAME NVARCHAR(50),
ADDRESS NVARCHAR(255)
)
CREATE TABLE LOANS(
LOANNUMBER VARCHAR(20) PRIMARY KEY,
AMOUNT FLOAT,
DATE DATE,
BRANCH NVARCHAR(100),
SSN VARCHAR(20) REFERENCES CUSTOMERS(SSN)
)
CREATE TABLE PAYMENTS(
PAYMENTNO VARCHAR(30),
AMOUNT FLOAT,
DATE DATE,
LOANNUMBER VARCHAR(20) REFERENCES LOANS(LOANNUMBER),
PRIMARY KEY(PAYMENTNO ,LOANNUMBER )
)
